const main = async () => {
	const sellAmount = parseUnits("1000", await usdc.read.decimals());

	const params = new URLSearchParams({
		originChainId: client.originChain.id,
		destinationChainId: client.destinationChain.id,
		originCurrency: usdc.address,
		destinationCurrency: weth.address,
		amount: sellAmount.toString(),
		senderAddress: client.senderAddress,
		receiverAddress: client.receiverAddress,
		tradeType: "EXACT_INPUT",
		integrator: client.integratorString,
		fee: client.fee
	});

	const quoteResponse = await fetch(
		"https://api.delora.build/v1/quotes" + params.toString(),
		{
			headers,
		}
	);

	const quote = await quoteResponse.json();
	console.log("Quote to swap 1000 USDC for WETH: ", quote);
	}
	main();
